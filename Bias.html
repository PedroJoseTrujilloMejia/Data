import polars as pl
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
import scipy.stats as stats
import pandas as pd
from ucimlrepo import fetch_ucirepo

# Fetch dataset
adult = fetch_ucirepo(id=2)
X = pl.from_pandas(adult.data.features)
y = pl.from_pandas(adult.data.targets)

# Combine features and target into a single Polars DataFrame
df = X.with_columns(y)

# The idea of this project is to analyze bias in artificial intelligence

# Convert necessary columns to Pandas for visualization
df_pandas = df.to_pandas()

# Function to compute proportion of >50K income in a given group
def compute_income_proportion(column):
    return df_pandas.groupby(column)['income'].value_counts(normalize=True).unstack()[">50K"]

# Analyze Income Distribution by Gender
gender_income = compute_income_proportion("sex")
print("Income Distribution by Gender:")
print(gender_income)

# Analyze Income Distribution by Race
race_income = compute_income_proportion("race")
print("Income Distribution by Race:")
print(race_income)

# Analyze Income Distribution by Education Level
education_income = compute_income_proportion("education")
print("Income Distribution by Education Level:")
print(education_income)

# Analyze Income Distribution by Occupation
occupation_income = compute_income_proportion("occupation")
print("Income Distribution by Occupation:")
print(occupation_income)

# Visualization: Bar plots for each demographic analysis
fig, axes = plt.subplots(2, 2, figsize=(15, 10))

# Gender
sns.barplot(x=gender_income.index, y=gender_income.values, ax=axes[0, 0], palette='coolwarm')
axes[0, 0].set_title("Income Distribution by Gender")
axes[0, 0].set_ylabel("Proportion earning >50K")

# Race
sns.barplot(x=race_income.index, y=race_income.values, ax=axes[0, 1], palette='coolwarm')
axes[0, 1].set_title("Income Distribution by Race")
axes[0, 1].set_ylabel("Proportion earning >50K")
axes[0, 1].set_xticklabels(axes[0, 1].get_xticklabels(), rotation=45)

# Education
sns.barplot(x=education_income.index, y=education_income.values, ax=axes[1, 0], palette='coolwarm')
axes[1, 0].set_title("Income Distribution by Education Level")
axes[1, 0].set_ylabel("Proportion earning >50K")
axes[1, 0].set_xticklabels(axes[1, 0].get_xticklabels(), rotation=45)

# Occupation
sns.barplot(x=occupation_income.index, y=occupation_income.values, ax=axes[1, 1], palette='coolwarm')
axes[1, 1].set_title("Income Distribution by Occupation")
axes[1, 1].set_ylabel("Proportion earning >50K")
axes[1, 1].set_xticklabels(axes[1, 1].get_xticklabels(), rotation=45)

plt.tight_layout()
plt.show()

# Statistical Testing: Chi-Square test for independence between categorical variables and income
def chi_square_test(column):
    contingency_table = pd.crosstab(df_pandas[column], df_pandas['income'])
    chi2, p, dof, expected = stats.chi2_contingency(contingency_table)
    return chi2, p

# Perform Chi-Square Tests
demographics = ['sex', 'race', 'education', 'occupation']
chi_square_results = {col: chi_square_test(col) for col in demographics}

# Print Results
for col, (chi2, p) in chi_square_results.items():
    print(f"Chi-Square Test for {col}:")
    print(f"Chi2 Statistic: {chi2:.4f}, P-Value: {p:.4f}")
    print("---")
